## Usage

```tsx
import {
  Accordion,
  AccordionItem,
  AccordionTrigger,
  AccordionPanel,
} from "@/components/ui/accordion";
```

```tsx
<Accordion>
  <AccordionItem>
    <AccordionTrigger>Is it accessible?</AccordionTrigger>
    <AccordionPanel>Yes. It adheres to the WAI-ARIA design pattern.</AccordionPanel>
  </AccordionItem>
</Accordion>
```
