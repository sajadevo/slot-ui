## Usage

```tsx
import { Avatar, AvatarImage, AvatarFallback } from "@/components/ui/avatar";
```

```tsx
<Avatar>
  <AvatarImage src="https://github.com/sajadevo.png" alt="@sajadevo" />
  <AvatarFallback>SA</AvatarFallback>
</Avatar>
```
