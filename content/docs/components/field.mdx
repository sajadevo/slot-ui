## Usage

```tsx
import {
  Field,
  FieldLabel,
  FieldDescription,
} from "@/components/ui/field";
import { Input } from "@/components/ui/input"
```

```tsx
<Field>
  <FieldLabel>Email</FieldLabel>
  <Input placeholder="someone@example.com" />
  <FieldDescription>We&apos;ll never share your email.</FieldDescription>
</Field>
```
