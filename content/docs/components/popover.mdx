## Usage

```tsx
import {
  Popover,
  PopoverTrigger,
  PopoverPopup,
  PopoverTitle,
  PopoverDescription
} from "@/components/ui/popover";
```

```tsx
<Popover>
  <PopoverTrigger>Open</PopoverTrigger>
  <PopoverPopup>
    <PopoverTitle>Title</PopoverTitle>
    <PopoverDescription>Description</PopoverDescription>
  </PopoverPopup>
</Popover>
```
