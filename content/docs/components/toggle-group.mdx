## Usage

```tsx
import { Toggle, ToggleGroup } from "@/components/ui/toggle-group";
```

```tsx
<ToggleGroup>
  <Toggle value="bold" aria-label="Toggle bold">
    <BoldIcon />
  </Toggle>
  <Toggle value="italic" aria-label="Toggle italic">
    <ItalicIcon />
  </Toggle>
  <Toggle value="underline" aria-label="Toggle underline">
    <UnderlineIcon />
  </Toggle>
</ToggleGroup>
```
